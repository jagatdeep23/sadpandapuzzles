private boolean addAndValidate (PuzzleWord word) {
    int dC = 0;
    int dR = 0;
    switch (word.getDirection ()) {
      case EAST:
        dC = 1;
        break;
      case SOUTH:
        dR = 1;
        break;
    }
    boolean crossed = false;
    int row = word.getRow ();
    int col = word.getColumn ();
    String w = word.getWord ();
    for (int i = 0; i < w.length (); i++) {
      PuzzleCell cell = matrix[col][row];
      char character = w.charAt (i);
      if (!cell.add (character)) {
        for (int j = i - 1; j >= 0; j--) {
          row -= dR;
          col -= dC;
          matrix[col][row].remove ();
        }
        return false;
      }
      firstWord = false;
      if (!crossed)
        crossed = (cell.getNumWords () > 1);
      row += dR;
      col += dC;
    }
    
    /* If word was succesfully added, but is not crossed and not the first word, remove */
    if (!crossed && !firstWord) {
      for (int i = 0; i < w.length (); i++) {
        System.out.println(matrix[col][row].getCharacter ());
        System.out.println("hello\n");
        row -= dR;
        col -= dC;
        matrix[col][row].remove();
      }
      return false;
    }
    return (true);
    
  }
